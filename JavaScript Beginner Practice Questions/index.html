Q21 Write a recursive function for factorial. ▾
A recursive function calls itself. Factorial: n! = n × (n-1) × … × 1. Base case: 0! = 1.
function factorial(n) {
  if (n === 0 || n === 1) return 1; // base case
  return n * factorial(n - 1);       // recursive call
}

console.log(factorial(5)); // 120  (5×4×3×2×1)
console.log(factorial(0)); // 1
Q22 Write recursive Fibonacci function. ▾
Fibonacci: each number = sum of the two before it. 0, 1, 1, 2, 3, 5, 8, 13…
function fibonacci(n) {
  if (n <= 0) return 0;
  if (n === 1) return 1;
  return fibonacci(n - 1) + fibonacci(n - 2);
}

console.log(fibonacci(6));  // 8
console.log(fibonacci(10)); // 55
Warning: Naive recursion is O(2ⁿ). Use memoization (Q25) for large n.

Q25 Write a function that memoizes factorial calculation. ▾
Memoization caches previous results so the same calculation isn't repeated.
function memoFactorial() {
  const cache = {};
  return function factorial(n) {
    if (n in cache) {
      console.log(`Cache hit: ${n}`);
      return cache[n];
    }
    if (n <= 1) return (cache[n] = 1);
    return (cache[n] = n * factorial(n - 1));
  };
}

const factorial = memoFactorial();
console.log(factorial(5));  // 120
console.log(factorial(5));  // Cache hit: 5 → 120

Q28 Create debounce function logic. ▾
Debounce delays a function call until after a set period of inactivity — useful for search inputs, resize events.
function debounce(fn, delay) {
  let timer;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

const handleSearch = debounce((query) => {
  console.log("Searching for:", query);
}, 500);

// Only fires 500ms after the last call
handleSearch("j");
handleSearch("ja");
handleSearch("jav");    // Only this fires
Q29 Create throttle function logic. ▾
Throttle ensures a function runs at most once per given time window — useful for scroll/resize handlers.
function throttle(fn, limit) {
  let inThrottle = false;
  return function(...args) {
    if (!inThrottle) {
      fn.apply(this, args);
      inThrottle = true;
      setTimeout(() => (inThrottle = false), limit);
    }
  };
}

const onScroll = throttle(() => {
  console.log("Scroll event");
}, 1000); // fires at most once per second

function once(fn) {

  return function(...args) {

      result = fn.apply(this, args);
    }
    return result;
  };
}

Q35 Explain the output of the test() function with a return before console.log. ▾
Code after a bare return statement is unreachable and never executes.
function test() {
  return;              // function ends here, returns undefined
  console.log("Hello"); // ❌ NEVER runs
}
console.log(test()); // undefined
Why? return; with no value explicitly returns undefined and exits the function immediately. Anything after it is dead code.

const ref       = original; // ← same reference, NOT a copy

ref.push(4);
ref[0] = 99;

console.log(original); // [99, 2, 3, 4] ← CHANGED!
console.log(ref);      // [99, 2, 3, 4] ← same object in memory

console.log(original === ref); // true — both point to same array
Key concept: Arrays (and objects) are stored by reference. Assigning const b = a does NOT copy — both variables point to the same memory location.
A19-H Hard: Create a true copy so the original array does not change. ▾ 
const original = [1, 2, 3, [4, 5]];

// ✅ Shallow copies (work for flat arrays):
const copy1 = [...original];          // spread
const copy2 = original.slice();       // slice with no args
const copy3 = Array.from(original);  // Array.from
const copy4 = [].concat(original);     // concat

copy1.push(99);
console.log(original); // [1,2,3,[4,5]] — untouched
console.log(copy1);   // [1,2,3,[4,5],99]

// ⚠️ Shallow copy problem with nested arrays:
copy1[3].push(6); // nested array is still shared!
console.log(original[3]); // [4,5,6] ← affected!

// ✅ Deep copy for nested arrays:
const deepCopy = JSON.parse(JSON.stringify(original));
// or: structuredClone(original)  ← modern, handles more types
deepCopy[3].push(99);
console.log(original[3]); // [4,5] — truly untouched
✨ 20. Spread Operator
A20-I Intermediate: Copy an array into a new array using spread operator. ▾ 
const original = [10, 20, 30];
const copy      = [...original];

copy.push(40);
console.log(original); // [10, 20, 30]  — unchanged
console.log(copy);     // [10, 20, 30, 40]

// Spread in function calls:
const nums = [3, 1, 4, 1, 5];
console.log(Math.max(...nums)); // 5  — spreads as individual args
A20-H Hard: Merge arrays and add extra values in between using spread. ▾ 
const a = [1, 2];
const b = [6, 7];

// Merge with extra values in between:
const merged = [...a, 3, 4, 5, ...b];
console.log(merged); // [1, 2, 3, 4, 5, 6, 7]

// Insert between with 3 arrays:
const start  = ["start"];
const middle = ["m1", "m2"];
const end    = ["end"];

const result = [...start, "before", ...middle, "after", ...end];
console.log(result);
// ["start","before","m1","m2","after","end"]

// Practical: insert element at specific position:
const arr = [1, 2, 5, 6];
const insertAt = 2;
const withInserted = [...arr.slice(0, insertAt), 3, 4, ...arr.slice(insertAt)];
console.log(withInserted); // [1, 2, 3, 4, 5, 6]
